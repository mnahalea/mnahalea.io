return shuffleArray([...whiteSubset, ...blackSubset]); } let testAudioQueue = balancedShuffle(); let audio = new Audio(); const startButton = document.getElementById("startButton"); if (startButton) { startButton.onclick = function () { startButton.style.display = "none"; } const nextAudio = testAudioQueue.shift(); currentAudioFile = nextAudio.audioFile; responseRecorded = false; audio.addEventListener('canplaythrough', () => { setTimeout(() => { startTime = Date.now();
